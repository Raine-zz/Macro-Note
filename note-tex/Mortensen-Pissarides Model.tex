\section{Basic Setting: Value Function}
	\begin{enumerate}
		\item For workers:
		\[
		W = w + \beta [(1-\delta)W + \delta U]
		\]
		\[
		U = b + \beta [f(\theta)W + (1 - f(\theta))U]
		\]
		\item For Firms
		\[
		F = (x - w) + \beta [(1 - \delta)F + \delta V]
		\]
		\[
		V = -c + \beta [q(\theta)F + (1 - q(\theta))V]
		\]
	\end{enumerate}

	Here, as zero-profit condition(V = 0), we have 
	\begin{align}
	&F = \frac{x - w}{1 - \beta(1 - \delta)}  = \frac{c}{\beta q(\theta)} \\
	&W = \frac{w + \delta\beta)U}{1 - \beta(1-\delta)} \nonumber \\
	&W - U = \frac{w - (1-\beta)U}{1 - \beta(1-\delta)} 
	\end{align}

	
\section{Optimal Wage}
	Wage w maximize Nash product:
	\[
	\max_w \; (W - U)^\gamma (F - V)^{1-\gamma}
	\]
	F.O.C.:
	\[
	(1-\gamma)(W - U) = \gamma F
	\]
	As equation (1) \& (2), we can rewrite the maximization function:
	\[
	\max_w \; \left[ \frac{w - (1-\beta)U}{1 - \beta(1-\delta)} \right]^\gamma 
	\left[ \frac{x - w}{1 - \beta(1-\delta)} \right]^{1-\gamma}
	\]
	F.O.C:
	\[
	\frac{\gamma}{w - (1-\beta)U} = \frac{1 - \gamma}{x - w}
	\]
	Which can be solved:
	\begin{equation}
		w = \gamma x + (1-\gamma)(1-\beta)U
	\end{equation}
	To calculate the unemployment value U:
	\begin{align*}
		&U = b + \beta \left[ f(\theta) W + (1 - f(\theta)) U \right] \\
		(1-\beta)&U = b + \beta f(\theta)(W - U) \\
		(1-\beta)&U = b + \beta f(\theta) \cdot \frac{\gamma}{1-\gamma} \cdot \frac{c}{\beta q(\theta)} \\
		(1-\beta)&U = b + \frac{\gamma}{1-\gamma} c \theta \\ 
	\end{align*}
	Finally, we can bring the U into equation(3):
	\begin{align*}
	w &= \gamma x + (1-\gamma) \left[ b + \frac{\gamma}{1-\gamma} c \theta \right] \\
	w &= (1-\gamma)b + \gamma x + \gamma c \theta
	\end{align*}
